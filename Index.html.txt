<script>
  // ── Categories ──
  const CATEGORIES = [
    { label: 'Top Stories', query: 'when:24h+allinurl:reuters.com' },
    { label: 'World',       query: 'when:24h+allinurl:reuters.com+world' },
    { label: 'US',          query: 'when:24h+allinurl:reuters.com+united+states' },
    { label: 'Business',    query: 'when:24h+allinurl:reuters.com+business' },
    { label: 'Markets',     query: 'when:24h+allinurl:reuters.com+markets' },
    { label: 'Technology',  query: 'when:24h+allinurl:reuters.com+technology' },
    { label: 'Politics',    query: 'when:24h+allinurl:reuters.com+politics' },
    { label: 'Science',     query: 'when:24h+allinurl:reuters.com+science' },
  ];

  // ── State ──
  let activeTab = 0;
  let cache = {};

  // ── Date header ──
  const dateEl = document.getElementById('header-date');
  const now = new Date();
  dateEl.innerHTML =
    now.toLocaleDateString('en-US', { weekday: 'long', month: 'long', day: 'numeric' }) +
    '<br>' +
    now.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' });

  // ── Build tabs ──
  const tabsEl = document.getElementById('tabs');
  CATEGORIES.forEach((cat, i) => {
    const btn = document.createElement('button');
    btn.className = 'tab' + (i === 0 ? ' active' : '');
    btn.textContent = cat.label;
    btn.addEventListener('click', () => selectTab(i));
    tabsEl.appendChild(btn);
  });

  function selectTab(i) {
    activeTab = i;
    document.querySelectorAll('.tab').forEach((t, j) => t.classList.toggle('active', j === i));
    loadFeed(i);
  }

  // ── RSS fetch via rss2json ──
  async function loadFeed(tabIndex) {
    const cat = CATEGORIES[tabIndex];
    const feedEl = document.getElementById('feed');
    const statusEl = document.getElementById('status');
    const statusText = document.getElementById('status-text');

    if (cache[tabIndex]) {
      renderArticles(cache[tabIndex], cat.label);
      return;
    }

    // Show skeleton
    setStatus('loading', 'Fetching latest wire...');
    feedEl.innerHTML = skeletonHTML(6);

    const rssUrl = `https://news.google.com/rss/search?q=${cat.query}&ceid=US:en&hl=en-US&gl=US`;
    const apiUrl = `https://api.rss2json.com/v1/api.json?rss_url=${encodeURIComponent(rssUrl)}&count=30`;

    try {
      const res = await fetch(apiUrl);
      const data = await res.json();

      if (!data.items || data.items.length === 0) {
        setStatus('error', 'No stories found');
        feedEl.innerHTML = '<p class="state-msg">No stories found — try another category.</p>';
        return;
      }

      cache[tabIndex] = data.items;
      renderArticles(data.items, cat.label);
      setStatus('live', `${data.items.length} stories · updated ${timeAgo(new Date())}`);

    } catch (err) {
      setStatus('error', 'Could not load feed');
      feedEl.innerHTML = '<p class="state-msg">Could not load feed. Check your connection.</p>';
    }
  }

  function renderArticles(items, categoryLabel) {
    const feedEl = document.getElementById('feed');
    feedEl.innerHTML = '';

    const list = document.createElement('div');
    list.className = 'article-list';

    items.forEach((item, i) => {
      const title = stripHTML(item.title || '');
      const snippet = stripHTML(item.description || item.content || '');
      const pubDate = item.pubDate ? new Date(item.pubDate) : null;
      const googleSearchUrl = `https://www.google.com/search?q=${encodeURIComponent(title)}&tbm=nws`;

      const el = document.createElement('div');
      el.className = 'article-item' + (i === 0 ? ' hero' : '');

      el.innerHTML = `
        <div class="article-body">
          <div class="article-category">${categoryLabel}</div>
          <div class="article-title">${title}</div>
          ${snippet ? `<div class="article-snippet">${snippet.slice(0, 180)}${snippet.length > 180 ? '…' : ''}</div>` : ''}
          <a class="read-btn" href="${googleSearchUrl}" target="_blank" rel="noopener noreferrer">
            <svg viewBox="0 0 10 10" fill="none" stroke="currentColor" stroke-width="1.5">
              <path d="M1 5h8M6 2l3 3-3 3"/>
            </svg>
            Read free coverage
          </a>
        </div>
        <div class="article-time">${pubDate ? timeAgo(pubDate) : ''}</div>
      `;

      list.appendChild(el);
    });

    feedEl.appendChild(list);
  }

  // ── Helpers ──
  function setStatus(type, msg) {
    const el = document.getElementById('status');
    const text = document.getElementById('status-text');
    el.className = 'status-bar ' + type;
    text.textContent = msg;
  }

  function stripHTML(str) {
    const tmp = document.createElement('div');
    tmp.innerHTML = str;
    return tmp.textContent || tmp.innerText || '';
  }

  function timeAgo(date) {
    const mins = Math.floor((Date.now() - date) / 60000);
    if (mins < 1) return 'just now';
    if (mins < 60) return `${mins}m ago`;
    const hrs = Math.floor(mins / 60);
    if (hrs < 24) return `${hrs}h ago`;
    return `${Math.floor(hrs / 24)}d ago`;
  }

  function skeletonHTML(n) {
    return `<div class="skeleton">${Array.from({length: n}, () => `
      <div class="skeleton-item">
        <div class="skel skel-cat"></div>
        <div class="skel skel-title"></div>
        <div class="skel skel-title-2"></div>
        <div class="skel skel-line"></div>
        <div class="skel skel-line-2"></div>
      </div>`).join('')}</div>`;
  }

  // ── Init ──
  loadFeed(0);
</script>
